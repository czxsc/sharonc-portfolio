<script>
    import { onDestroy } from 'svelte';
    import {
        pitchCurrent,
        rollCurrent,
        yawCurrent,
        throttleCurrent,
        pitchMax,
        rollMax,
        yawMax,
        throttleMax,
        pitchMin,
        rollMin,
        yawMin,
        throttleMin,
        pitchReverse,
        rollReverse,
        yawReverse,
        throttleReverse,
        radio5,
        radio6,
        radio7,
        radio8,
        radio9,
        radio10,
        radio11,
        radio12,
        radio13,
        radio14
    } from '../stores/DpkStore.js';
    import {
        startRadioCalibration,
        finishRadioCalibration
    } from './windows/DpkActions.js';

    export let calibrationInfo =
        'Move each transmitter stick through its full range. The green fill shows the live input and the red markers indicate the captured calibration limits.';

    const SIGNAL_MIN = 800;
    const SIGNAL_MAX = 2200;
    let calibrationRunning = false;
    let lastSavedSnapshot = null;

    function clamp(value, min, max) {
        return Math.min(Math.max(value, min), max);
    }

    function toPercent(value) {
        const safeValue = Number(value) || 0;
        return ((clamp(safeValue, SIGNAL_MIN, SIGNAL_MAX) - SIGNAL_MIN) / (SIGNAL_MAX - SIGNAL_MIN)) * 100;
    }

    function displayValue(value) {
        const numeric = Number(value);
        return Number.isFinite(numeric) ? Math.round(numeric) : 0;
    }

    let values = {};
    const unsubscribers = [];

    function bindStore(key, store) {
        unsubscribers.push(
            store.subscribe((value) => {
                values = { ...values, [key]: value };
            })
        );
    }

    [
        ['pitchCurrent', pitchCurrent],
        ['rollCurrent', rollCurrent],
        ['yawCurrent', yawCurrent],
        ['throttleCurrent', throttleCurrent],
        ['pitchMax', pitchMax],
        ['rollMax', rollMax],
        ['yawMax', yawMax],
        ['throttleMax', throttleMax],
        ['pitchMin', pitchMin],
        ['rollMin', rollMin],
        ['yawMin', yawMin],
        ['throttleMin', throttleMin],
        ['pitchReverse', pitchReverse],
        ['rollReverse', rollReverse],
        ['yawReverse', yawReverse],
        ['throttleReverse', throttleReverse],
        ['radio5', radio5],
        ['radio6', radio6],
        ['radio7', radio7],
        ['radio8', radio8],
        ['radio9', radio9],
        ['radio10', radio10],
        ['radio11', radio11],
        ['radio12', radio12],
        ['radio13', radio13],
        ['radio14', radio14]
    ].forEach(([key, store]) => bindStore(key, store));

    onDestroy(() => {
        unsubscribers.forEach((unsubscribe) => unsubscribe());
    });

    const primaryChannels = [
        {
            label: 'Roll',
            orientation: 'horizontal',
            currentKey: 'rollCurrent',
            minKey: 'rollMin',
            maxKey: 'rollMax',
            reverseKey: 'rollReverse'
        },
        {
            label: 'Pitch',
            orientation: 'vertical',
            currentKey: 'pitchCurrent',
            minKey: 'pitchMin',
            maxKey: 'pitchMax',
            reverseKey: 'pitchReverse'
        },
        {
            label: 'Throttle',
            orientation: 'vertical',
            currentKey: 'throttleCurrent',
            minKey: 'throttleMin',
            maxKey: 'throttleMax',
            reverseKey: 'throttleReverse'
        },
        {
            label: 'Yaw',
            orientation: 'horizontal',
            currentKey: 'yawCurrent',
            minKey: 'yawMin',
            maxKey: 'yawMax',
            reverseKey: 'yawReverse'
        }
    ];

    const auxChannels = [
        { label: 'Radio 5', key: 'radio5' },
        { label: 'Radio 6', key: 'radio6' },
        { label: 'Radio 7', key: 'radio7' },
        { label: 'Radio 8', key: 'radio8' },
        { label: 'Radio 9', key: 'radio9' },
        { label: 'Radio 10', key: 'radio10' },
        { label: 'Radio 11', key: 'radio11' },
        { label: 'Radio 12', key: 'radio12' },
        { label: 'Radio 13', key: 'radio13' },
        { label: 'Radio 14', key: 'radio14' }
    ];

    function buildCalibrationSnapshot() {
        return {
            roll: {
                current: displayValue(values.rollCurrent),
                min: displayValue(values.rollMin),
                max: displayValue(values.rollMax),
                reverse: Boolean(values.rollReverse)
            },
            pitch: {
                current: displayValue(values.pitchCurrent),
                min: displayValue(values.pitchMin),
                max: displayValue(values.pitchMax),
                reverse: Boolean(values.pitchReverse)
            },
            throttle: {
                current: displayValue(values.throttleCurrent),
                min: displayValue(values.throttleMin),
                max: displayValue(values.throttleMax),
                reverse: Boolean(values.throttleReverse)
            },
            yaw: {
                current: displayValue(values.yawCurrent),
                min: displayValue(values.yawMin),
                max: displayValue(values.yawMax),
                reverse: Boolean(values.yawReverse)
            },
            auxiliary: Object.fromEntries(
                auxChannels.map((channel) => [channel.key, displayValue(values[channel.key])])
            )
        };
    }

    function handleCalibrationButton() {
        if (!calibrationRunning) {
            startRadioCalibration();
            calibrationRunning = true;
            return;
        }

        lastSavedSnapshot = buildCalibrationSnapshot();
        finishRadioCalibration(lastSavedSnapshot);
        calibrationRunning = false;
    }
  </script>

const postParamValue = async (name, value) => {
    const pname = name ?? '';
    const pvalue =
        value !== undefined && value !== null ? String(value).trim() : '';

    if (!pname) {
        return {ok: false, status: 0, error: 'Missing param name'};
    }
    if (pvalue === '') {
        return {ok: false, status: 0, error: `Missing value for ${pname}`};
    }

    const res = await fetch(API_URL + `/param_set`, {
        method: 'POST',
        mode: 'cors',
        headers: {
            'Content-Type': 'application/json'
        },
        body: JSON.stringify({
            pname,
            value: pvalue
        })
    });

    let data = null;
    try {
        data = await res.json();
    } catch {
        // backend currently returns text for this endpoint
    }

    return {ok: res.ok, status: res.status, data};
};

export const sendParam = parameter => {
    const name = parameter?.name ?? parameter?.pname;
    const value = parameter?.value;

    postParamValue(name, value)
        .then(result => {
            if (result.ok) {
                createAlert.success(`Set param ${name} to ${value}`);
            } else {
                createAlert.error(`Failed to set param ${name}`);
            }
        })
        .catch(() => {
            createAlert.error(`Failed to set param ${name}`);
        });
};

export const sendParamBatch = async (parameters, options = {}) => {
    const delayMs = options.delayMs ?? 0;
    const showSummaryAlert = options.showSummaryAlert ?? false;

    const failed = [];
    const succeeded = [];

    for (const parameter of parameters) {
        const name = parameter?.name ?? parameter?.pname;
        const value = parameter?.value;
        try {
            const result = await postParamValue(name, value);
            if (result.ok) {
                succeeded.push({name, value});
            } else {
                failed.push({
                    name,
                    value,
                    status: result.status,
                    data: result.data,
                    error: result.error
                });
            }
        } catch (error) {
            failed.push({name, value, error: String(error)});
        }

        if (delayMs) {
            await new Promise(resolve => setTimeout(resolve, delayMs));
        }
    }

    if (showSummaryAlert) {
        if (failed.length > 0) {
            createAlert.error(
                `Failed to set ${failed.length} parameter(s): ${failed
                    .map(item => `${item.name}=${item.value}`)
                    .join(', ')}`
            );
        } else {
            createAlert.success(
                `Successfully set ${succeeded.length} parameter(s).`
            );
        }
    }

    return {ok: failed.length === 0, failed, succeeded};
};

const MS_TO_KT = 1.943844;
const M_TO_FT = 3.28084;

function toNumberLoose(x) {
    if (typeof x === 'number') return x;
    if (typeof x !== 'string') return NaN;
    const m = x.match(/[+-]?\d*\.?\d+(?:[eE][+-]?\d+)?/);
    return m ? Number(m[0]) : NaN;
}

function fmtNumber(n, digits = 2) {
    return Number.isFinite(n) ? n.toFixed(digits) : String(n);
}

function formatByKind(kind, raw) {
    const n = toNumberLoose(raw);

    switch (kind) {
        case 'alt': {
            const ft = n * M_TO_FT;
            return `${fmtNumber(ft, 2)} ft  ${fmtNumber(n, 2)} m`;
        }
        case 'speed': {
            const kt = n * MS_TO_KT;
            return `${fmtNumber(kt, 2)} kt  ${fmtNumber(n, 2)} m/s`;
        }
        case 'mps':
            return `${fmtNumber(n, 2)} m/s`;
        case 'deg':
            return `${fmtNumber(n, 2)} °`;
        case 'radps':
            return `${fmtNumber(n, 4)} rad/s`;
        case 'pct01':
            return `${fmtNumber(n * 100, 0)} %`;
        case 'pct0_100':
            return `${fmtNumber(n, 0)} %`;
        case 'sec':
            return `${fmtNumber(n, 0)} s`;
        case 'm':
            return `${fmtNumber(n, 0)} m`;
        case 'V':
            return `${fmtNumber(n, 2)} V`;
        case 'A':
            return `${fmtNumber(n, 2)} A`;
        case 'mAh':
            return `${fmtNumber(n, 0)} mAh`;
        case 'int':
            return `${fmtNumber(n, 0)}`;
        case 'bool':
            return raw ? 'Yes' : 'No';
        case 'text':
            return String(raw);
        default:
            return String(raw);
    }
}

function findFieldSpec(prompt) {
    const p = prompt.toLowerCase();
    const ordered = [...FIELDS].sort(
        (a, b) => (b.keys[0]?.length || 0) - (a.keys[0]?.length || 0)
    );
    for (const spec of ordered) {
        if (spec.keys.some(k => p.includes(k))) return spec;
    }
    return null;
}

function formatResponse(prompt, botValue) {
    const spec = findFieldSpec(prompt);
    const value = botValue
        ?.toString()
        .trim()
        .replace(/^[\s"{[]+|[\s"}\]]+$/g, '');
    if (!spec) {
        return `${prompt}: ${value}`;
    }
    const pretty = formatByKind(spec.fmt, value);
    return `${spec.label}: ${pretty}`;
}
